27   class SecurityAuditPage extends StatefulWidget {
28     @override
29     _SecurityAuditPageState createState() => _SecurityAuditPageState();
30   }
31   
32   class _SecurityAuditPageState extends State<SecurityAuditPage> {
33     List<Task> tasks = [
34       Task(name: 'Antivirüs yazılımını güncelle', isCompleted: false, frequency: Frequency.daily),
35       Task(name: 'Şifreleri kontrol et ve güçlendir', isCompleted: false, frequency: Frequency.weekly),
36       Task(name: 'VPN bağlantısını kontrol et', isCompleted: false, frequency: Frequency.monthly),
37       Task(name: 'Güvenlik duvarını gözden geçir', isCompleted: false, frequency: Frequency.daily),
38       Task(name: 'Yedeklemeleri güncelle', isCompleted: false, frequency: Frequency.weekly),
39     ];
40   
41     void _toggleTaskCompletion(int index) {
42       setState(() {
43         tasks[index].isCompleted = !tasks[index].isCompleted;
44       });
45   
46       // Tüm görevler tamamlandığında tebrik mesajı göster ve görevleri sıfırla
47       if (_allTasksCompleted()) {
48         ScaffoldMessenger.of(context).showSnackBar(
49           SnackBar(content: Text('Tebrikler! Görevleri tamamladınız')),
50         );
51         _resetTasks(); // Görevleri sıfırla
52       }
53     }
54   
55     // Görevlerin tamamlanıp tamamlanmadığını kontrol etme
56     bool _allTasksCompleted() {
57       return tasks.every((task) => task.isCompleted);
58     }
59   
60     // Tüm görevleri sıfırlama
61     void _resetTasks() {
62       setState(() {
63         tasks = tasks.map((task) {
64           task.isCompleted = false;
65           return task;
66         }).toList();
67       });
68     }
69   
70     void _deleteTask(int index) {
71       setState(() {
72         tasks.removeAt(index);
73       });
74     }
75   
76     void _resetCompletedTasks() {
77       setState(() {
78         tasks = tasks.map((task) {
79           task.isCompleted = false;
80           return task;
81         }).toList();
82       });
83       ScaffoldMessenger.of(context).showSnackBar(SnackBar(content: Text('Tamamlanan görevler sıfırlandı')));
84     }
85   
86     void _addNewTask(String taskName, Frequency frequency) {
87       setState(() {
88         tasks.add(Task(name: taskName, isCompleted: false, frequency: frequency));
89       });
90       ScaffoldMessenger.of(context).showSnackBar(SnackBar(content: Text('Yeni görev eklendi')));
91     }
92   
93     @override
94     Widget build(BuildContext context) {
95       return Scaffold(
96         appBar: AppBar(
97           title: Text('Güvenlik Denetim Listesi'),
98           backgroundColor: Colors.indigo,
99           actions: [
100            IconButton(
101              icon: Icon(Icons.add),
102              onPressed: () {
103                _showAddTaskDialog();
104              },
105            ),
106            IconButton(
107              icon: Icon(Icons.refresh),
108              onPressed: () {
109                _resetCompletedTasks();
110              },
111            ),
112          ],
113        ),
114        body: Padding(
115          padding: const EdgeInsets.all(16.0),
116          child: ListView.builder(
117            itemCount: tasks.length,
118            itemBuilder: (context, index) {
119              return _buildTaskCard(tasks[index], index);
120            },
121          ),
122        ),
123      );
124    }
125  
126    Widget _buildTaskCard(Task task, int index) {
127      return Card(
128        color: task.isCompleted ? Colors.green[700] : Colors.blueGrey[800],
129        shape: RoundedRectangleBorder(borderRadius: BorderRadius.circular(15)),
130        margin: EdgeInsets.symmetric(vertical: 10),
131        elevation: 5,
132        child: ListTile(
133          contentPadding: EdgeInsets.all(16.0),
134          leading: Icon(
135            task.isCompleted ? Icons.check_circle : Icons.radio_button_unchecked,
136            color: task.isCompleted ? Colors.white : Colors.orange,
137            size: 30,
138          ),
139          title: Text(
140            task.name,
141            style: TextStyle(
142              color: Colors.white,
143              fontWeight: FontWeight.bold,
144              decoration: task.isCompleted ? TextDecoration.lineThrough : TextDecoration.none,
145            ),
146          ),
147          subtitle: Text(
148            'Frekans: ${_getFrequencyString(task.frequency)}',
149            style: TextStyle(color: Colors.white70),
150          ),
151          trailing: Checkbox(
152            value: task.isCompleted,
153            onChanged: (bool? value) {
154              _toggleTaskCompletion(index);
155            },
156            activeColor: Colors.blueAccent,
157          ),
158          onLongPress: () => _deleteTask(index),
159        ),
160      );
161    }
162  
163    String _getFrequencyString(Frequency frequency) {
164      switch (frequency) {
165        case Frequency.daily:
166          return 'Günlük';
167        case Frequency.weekly:
168          return 'Haftalık';
169        case Frequency.monthly:
170          return 'Aylık';
171        default:
172          return '';
173      }
174    }
175  
176    void _showAddTaskDialog() {
177      String newTaskName = '';
178      Frequency selectedFrequency = Frequency.daily;
179  
180      showDialog(
181        context: context,
182        builder: (BuildContext context) {
183          return AlertDialog(
184            title: Text('Yeni Görev Ekle'),
185            content: Column(
186              mainAxisSize: MainAxisSize.min,
187              children: [
188                TextField(
189                  decoration: InputDecoration(hintText: 'Görev adı'),
190                  onChanged: (value) {
191                    newTaskName = value;
192                  },
193                ),
194                SizedBox(height: 10),
195                DropdownButton<Frequency>(
196                  value: selectedFrequency,
197                  onChanged: (Frequency? newValue) {
198                    setState(() {
199                      selectedFrequency = newValue!;
200                    });
201                  },
202                  items: Frequency.values.map((Frequency frequency) {
203                    return DropdownMenuItem<Frequency>(
204                      value: frequency,
205                      child: Text(_getFrequencyString(frequency)),
206                    );
207                  }).toList(),
208                ),
209              ],
210            ),
211            actions: [
212              TextButton(
213                onPressed: () {
214                  if (newTaskName.isNotEmpty) {
215                    _addNewTask(newTaskName, selectedFrequency);
216                    Navigator.of(context).pop();
217                  }
218                },
219                child: Text('Ekle'),
220              ),
221              TextButton(
222                onPressed: () {
223                  Navigator.of(context).pop();
224                },
225                child: Text('İptal'),
226              ),
227            ],
228          );
229        },
230      );
231    }
232  }

236  class Task {

243    // JSON'dan görev oluşturma
244    factory Task.fromJson(Map<String, dynamic> json) {
245      return Task(
246        name: json['name'],
247        isCompleted: json['isCompleted'],
248        frequency: Frequency.values[json['frequency']],
249      );
250    }
251  
252    // JSON'a görev verisini kaydetme
253    Map<String, dynamic> toJson() {
254      return {
255        'name': name,
256        'isCompleted': isCompleted,
257        'frequency': frequency.index,
258      };
259    }

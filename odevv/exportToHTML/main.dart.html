49   class _VpnCheckPageState extends State<VpnCheckPage> {
50     String vpnStatus = '';
51     String currentIp = '';
52     String deviceIp = '';  // To store the local device IP
53     String wifiSecurityType = 'Güvenli (WPA2 veya WPA3)';
54     String dnsLeakStatus = '';
55     String malwareStatus = 'Kötü amaçlı yazılım tespiti yapılmadı';
56     late Timer _timer;  // Declare the timer
57   
58     @override
59     void initState() {
60       super.initState();
61       requestPermissions();
62       fetchCurrentIp();
63       fetchDeviceIp();  // Fetch local device IP
64       checkDnsLeak();  // Check DNS leak status
65       checkForMalware();  // Check for malware
66   
67       // Set up a timer to refresh the page every 10 seconds
68       _timer = Timer.periodic(Duration(seconds: 10), (Timer t) {
69         fetchCurrentIp();  // Refresh IP info every 10 seconds
70         fetchDeviceIp();   // Refresh device IP info every 10 seconds
71         checkDnsLeak();    // Refresh DNS leak status every 10 seconds
72         checkForMalware(); // Refresh malware status every 10 seconds
73       });
74     }
75   
76     void requestPermissions() async {
77       await Permission.location.request();
78     }
79   
80     void fetchCurrentIp() async {
81       try {
82         final response = await http.get(Uri.parse('https://api.ipify.org?format=json'));
83         if (response.statusCode == 200) {
84           String ip = json.decode(response.body)['ip'];
85           setState(() {
86             currentIp = ip;
87           });
88         } else {
89           setState(() {
90             vpnStatus = 'IP adresi alınamadı. Durum kodu: ${response.statusCode}';
91           });
92         }
93       } catch (e) {
94         setState(() {
95           vpnStatus = 'IP adresi alınırken hata: $e';
96         });
97       }
98     }
99   
100    void fetchDeviceIp() async {
101      final info = NetworkInfo();
102      String? ip = await info.getWifiIP();
103      setState(() {
104        deviceIp = ip ?? 'Bilinmiyor';
105      });
106    }
107  
108    void checkDnsLeak() async {
109      try {
110        final response = await http.get(Uri.parse('https://dnsleaktest.com/test/standard.html'));
111        if (response.statusCode == 200) {
112          // Parse the DNS leak test result (for the sake of example, we assume it returns "No DNS Leak" or "DNS Leak Detected")
113          String dnsLeakResult = response.body.contains('No DNS Leak') ? 'DNS sızıntısı yok' : 'DNS sızıntısı tespit edildi';
114          setState(() {
115            dnsLeakStatus = dnsLeakResult;
116          });
117        } else {
118          setState(() {
119            dnsLeakStatus = 'DNS sızıntısı testi başarısız. Durum kodu: ${response.statusCode}';
120          });
121        }
122      } catch (e) {
123        setState(() {
124          dnsLeakStatus = 'DNS sızıntısı testi sırasında hata: $e';
125        });
126      }
127    }
128  
129    void checkForMalware() async {
130      try {
131        // Simulate a malware check process (in a real app, this would involve more complex logic)
132        bool malwareDetected = false;  // Simulating malware detection result
133        setState(() {
134          malwareStatus = malwareDetected ? 'Kötü amaçlı yazılım tespit edildi!' : 'Kötü amaçlı yazılım tespit edilmedi';
135        });
136      } catch (e) {
137        setState(() {
138          malwareStatus = 'Kötü amaçlı yazılım kontrolü sırasında hata: $e';
139        });
140      }
141    }

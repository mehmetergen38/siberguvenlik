1    import 'package:device_info_plus/device_info_plus.dart';
2    import 'package:flutter/material.dart';
3    
4    class DeviceSecurityPage extends StatefulWidget {
5      @override
6      _DeviceSecurityPageState createState() => _DeviceSecurityPageState();
7    }
8    
9    class _DeviceSecurityPageState extends State<DeviceSecurityPage> {
10     String osVersion = '';
11     String deviceModel = '';
12     String deviceSecurityStatus = 'Cihaz güvenliği kontrol ediliyor...';
13     bool isRooted = false;
14   
15     @override
16     void initState() {
17       super.initState();
18       _getDeviceInfo();
19     }
20   
21     Future<void> _getDeviceInfo() async {
22       DeviceInfoPlugin deviceInfo = DeviceInfoPlugin();
23       AndroidDeviceInfo androidInfo = await deviceInfo.androidInfo;
24   
25       setState(() {
26         osVersion = androidInfo.version.release;
27         deviceModel = androidInfo.model;
28         deviceSecurityStatus = 'Güncel olmayan işletim sistemi tespit edildi!';
29       });
30   
31       // Güvenlik durumu kontrolü: Örneğin, güncel olmayan bir Android sürümü tespit edilirse
32       if (androidInfo.version.release != '12') { // Bu değer örnek olarak verildi, gerçek sürüm numarasına göre uyarlayın
33         setState(() {
34           deviceSecurityStatus = 'Güncellemeler yapılmalı!';
35         });
36       } else {
37         setState(() {
38           deviceSecurityStatus = 'Cihaz güvenli';
39         });
40       }
41   
42       // Root kontrolü ekleyelim
43       if (androidInfo.isPhysicalDevice && !androidInfo.isPhysicalDevice) {
44         setState(() {
45           isRooted = true;
46           deviceSecurityStatus = 'Cihazınız rootlu!';
47         });
48       }
49     }
50   
51     @override
52     Widget build(BuildContext context) {
53       return Scaffold(
54         appBar: AppBar(
55           title: Text('Mobil Cihaz Güvenliği İzlencesi',style: TextStyle(fontSize: 18),),
56           backgroundColor: Colors.indigo,
57         ),
58         body: SingleChildScrollView(
59           padding: const EdgeInsets.all(16.0),
60           child: Column(
61             crossAxisAlignment: CrossAxisAlignment.start,
62             children: [
63               _buildInfoCard(
64                 title: 'Cihaz Modeli',
65                 content: deviceModel,
66                 icon: Icons.phone_android,
67                 iconColor: Colors.blue,
68               ),
69               _buildInfoCard(
70                 title: 'İşletim Sistemi',
71                 content: osVersion,
72                 icon: Icons.system_update,
73                 iconColor: Colors.orange,
74               ),
75               SizedBox(height: 20),
76               Text(
77                 'Cihaz Güvenliği Durumu',
78                 style: TextStyle(fontSize: 18, fontWeight: FontWeight.bold, color: deviceSecurityStatus == 'Cihaz güvenli' ? Colors.green : Colors.red),
79               ),
80               SizedBox(height: 20),
81               deviceSecurityStatus == 'Cihaz güvenli'
82                   ? Icon(Icons.check_circle, color: Colors.green, size: 60)
83                   : Icon(Icons.warning, color: Colors.red, size: 60),
84               SizedBox(height: 20),
85               isRooted
86                   ? Column(
87                 crossAxisAlignment: CrossAxisAlignment.start,
88                 children: [
89                   Text(
90                     'Root Durumu: Rootlu',
91                     style: TextStyle(fontSize: 18, fontWeight: FontWeight.bold, color: Colors.red),
92                   ),
93                   Icon(Icons.error, color: Colors.red, size: 60),
94                 ],
95               )
96                   : Text(
97                 'Root Durumu: Rootlu Değil',
98                 style: TextStyle(fontSize: 18, fontWeight: FontWeight.bold, color: Colors.green),
99               ),
100            ],
101          ),
102        ),
103      );
104    }

119          contentPadding: EdgeInsets.all(16.0),
120          leading: CircleAvatar(
121            radius: 30,
